Events + Run Inline JS

                          <button data-swiss="event(myCustomEvent)">
                            Dispatch custom event
                          </button>
                          <script>
                            document.addEventListener('myCustomEvent', () => {
                              alert('”myCustomEvent” triggered by Swiss.js!')
                            });
                          </script>
